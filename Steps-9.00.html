Measures Calculation


CasesPerYear = Calculate(DIVIDE(SUM('BYAREA1'[COUNT]), DISTINCTCOUNT(Date1[Year]),0)+0,'BYAREA1'[EVENT_TYPE] = "Incidence")

DeathPearYear = Calculate(DIVIDE(SUM('BYAREA1'[COUNT]), DISTINCTCOUNT(Date1[Year]),0)+0,'BYAREA1'[EVENT_TYPE] = "Mortality")


Mortality Rate =

CALCULATE(DIVIDE(SUM('BYAREA1'[COUNT]),SUM('BYAREA1'[POPULATION])) , 'BYAREA1'[EVENT_TYPE] = "Mortality") * 100000



Incidence Rate =

CALCULATE(DIVIDE(SUM('BYAREA1'[COUNT]),SUM('BYAREA1'[POPULATION])), 'BYAREA1'[EVENT_TYPE] = "Incidence") * 100000


YOY Incidence =

VAR __PREV_yr = CALCULATE ([Incidence Rate], DATEADD(Date1[Date], -1, YEAR))
                 RETURN
DIVIDE ([Incidence Rate] - __PREV_yr, __PREV_yr) +0

YOY Mortality =

VAR __PRE_YR = CALCULATE ([Mortality Rate], DATEADD(Date1[Date], -1, YEAR))
                 RETURN
DIVIDE ([Mortality Rate] - __PRE_YR, __PRE_YR) +0

Incidence coloring =

IF ([YOY Incidence] >0,1,2)


Mortality coloring = IF ([YOY Mortality] > 0,1,2)


Indicator Incidence = If ([YOY Incidence]<0,UNICHAR(9660),UNICHAR(9650))

Indicator Mortality = If ([YOY Mortality] < 0,UNICHAR(9660),UNICHAR(9650))
